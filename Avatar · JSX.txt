// src/components/Avatar.jsx
import { avatarColor, initials } from "../utils/helpers.js";

export default function Avatar({ name = "", size = "md", className = "" }) {
  const sizeMap = { sm: "w-7 h-7 text-xs", md: "w-9 h-9 text-sm", lg: "w-11 h-11 text-base" };
  return (
    <div className={`${sizeMap[size]} ${avatarColor(name)} rounded-full flex items-center justify-center text-white font-display font-bold flex-shrink-0 ${className}`}>
      {initials(name)}
    </div>
  );
}
